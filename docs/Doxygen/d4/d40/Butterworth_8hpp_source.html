22 SOSCoefficients<T, (N + 1) / 2> butter_coeff(double f_c) {
23  const T gamma = 1 / std::tan(M_PI * f_c / 2); // pre-warp factor
24  auto make_sos = [=](uint8_t k) {
25  const T gamma2 = gamma * gamma;
26  const T alpha = 2 * std::cos(2 * M_PI * (2 * k + N + 1) / (4 * N));
27  return BiQuadCoefficients<T>{
28  {{1, 2, 1}}, // b0, b1, b2
29  {{
30  gamma2 - alpha * gamma + 1, // a0
31  2 * (1 - gamma2), // a1
32  gamma2 + alpha * gamma + 1, // a2
33  }},
34  };
35  };
36 
37  auto make_fos = [=]() {
38  return BiQuadCoefficients<T>{
39  {{1, 1}}, // b0, b1
40  {{
41  gamma + 1, // a0
42  1 - gamma, // a1
43  }},
44  };
45  };

48 
49  if (N % 2 == 0) {

52  } else {
