10 /**
11  * @brief BiQuad filter Direct Form 1 implementation that normalizes the
12  * coefficients upon initialization.
13  *
14  * This class is faster than @ref NonNormalizingBiQuadFilterDF1, because each
15  * filter iteration involves only addition and multiplication, no divisions.
16  * It works great for floating point numbers, but might be less ideal
17  * for integer types, because it can create large rounding errors on the
18  * coefficients.
19  *
20  * Implements the following difference equation:
21  *
22  * @f[
23  * y[n] = \frac{b_0 \cdot x[n] + b_1 \cdot x[n-1] + b_2 \cdot x[n-2]
24  * - a_1 \cdot y[n-1] - a_2 \cdot y[n-2]}{a_0}
25  * @f]
26  */
27 template <class T>
28 class NormalizingBiQuadFilterDF1 {
29  public:
30  NormalizingBiQuadFilterDF1() = default;
31 
32  NormalizingBiQuadFilterDF1(const AH::Array<T, 3> &b,
33  const AH::Array<T, 3> &a)
34  : b{b / a[0]}, a{a.template slice<1, 3>() / a[0]} {}
35 
36  NormalizingBiQuadFilterDF1(const AH::Array<T, 3> &b,
37  const AH::Array<T, 3> &a, T gain)
38  : b{b * gain / a[0]}, a{a.template slice<1, 3>() / a[0]} {}
39 
40  static inline T update(T input, AH::Array<T, 2> &x, AH::Array<T, 2> &y,
41  const AH::Array<T, 3> &b, const AH::Array<T, 2> &a) {
42  T b_terms = input * b[0] + x[0] * b[1] + x[1] * b[2];
43  T a_terms = y[0] * a[0] + y[1] * a[1];
44  x[1] = x[0];
45  x[0] = input;
46  y[1] = y[0];
47  y[0] = b_terms - a_terms;
48  return y[0];
49  }
50

53  * return the new output @f$ y[n] @f$.
54  *
55  * @param input
56  * The new input @f$ x[n] @f$.
57  * @return The new output @f$ y[n] @f$.
58  */
59  T operator()(T input) { return update(input, x, y, b, a); }
60
